028    package com.edugility.throwables;
029    
030    import java.util.ArrayList;
031    import java.util.Iterator;
032    import java.util.List;
033    
034    import org.junit.Before;
035    import org.junit.Test;
036    
037    import static org.junit.Assert.*;
038    
039    public class TestCaseThrowables {
040    
041      private Throwable first;
042    
043      private Throwable thirdToLast;
044    
045      private Throwable secondToLast;
046    
047      private Throwable last;
048    
049      public TestCaseThrowables() {
050        super();
051      }
052    
053      @Before
054      public void setUp() {
055        this.last = new NumberFormatException();
056        this.secondToLast = new IllegalArgumentException(this.last);
057        this.thirdToLast = new IllegalArgumentException(this.secondToLast);
058        this.first = new Throwable(this.thirdToLast);
059      }
060    
061      @Test
062      public void testFirstInstance() {
063        final NumberFormatException result = Throwables.firstInstance(this.first, NumberFormatException.class);
064        assertSame(result, this.last);
065      }
066    
067      @Test
068      public void testLastInstance() {
069        final IllegalArgumentException result = Throwables.lastInstance(this.first, IllegalArgumentException.class);
070        assertSame(result, this.last);
071      }
072    
073      @Test
074      public void testGetRootCause() {
075        Throwable cause = Throwables.getRootCause(this.first);
076        assertSame(cause, this.last);
077        cause = Throwables.getRootCause(null);
078        assertNull(cause);
079      }
080    
081      @Test
082      public void testToList() {
083        List<Throwable> list = Throwables.toList(null);
084        assertNotNull(list);
085        assertTrue(list.isEmpty());
086    
087        list = Throwables.toList(this.first);
088        assertNotNull(list);
089        assertEquals(4, list.size());
090        assertSame(this.first, list.get(0));
091        assertSame(this.thirdToLast, list.get(1));
092        assertSame(this.secondToLast, list.get(2));
093        assertSame(this.last, list.get(3));
094      }
095    
096      @Test
097      public void testToListWithThrowableChain() {
098        final ThrowableChain chain = new ThrowableChain();
099    
100        // A ThrowableChain always contains itself.
101        assertEquals(1, chain.size());
102    
103        // Setting the cause does nothing to the list of affiliates or a
104        // ThrowableChain's size.
105        final Exception cause = new Exception("cause");
106        chain.add(cause); // actually initializes cause, does not add to list
107        assertEquals(1, chain.size());
108        assertSame(cause, chain.getCause());
109    
110        // Adding an affiliate to a ThrowableChain with a cause results in
111        // size 2, not 3.
112        final Exception firstAffiliate = new Exception("firstAffiliate");
113        chain.add(firstAffiliate);
114        assertEquals(2, chain.size());
115    
116        // But asking Throwables#toList() to do its thing WILL produce a
117        // list of size 3, since causes ARE considered.
118        final List<Throwable> list = Throwables.toList(chain);
119        assertNotNull(list);
120        assertEquals(3, list.size());
121    
122        // A ThrowableChain is always the first element in the list that
123        // results from Throwables#toList() when it is passed to it.
124        assertSame(chain, list.get(0));
125        
126        // Causes are next (depth first).
127        assertSame(cause, list.get(1));
128    
129        // After one list element's causal chain is iterated, then the
130        // next one is iterated.
131        assertSame(firstAffiliate, list.get(2));
132      }
133    
134      @Test
135      public void testWeird() {
136        // Tests a throwable that is an Iterable<Throwable> that does NOT
137        // contain itself
138        final ThrowableCollection c = new ThrowableCollection();
139        final Iterator<Throwable> i = c.iterator();
140        assertNotNull(i);
141        assertTrue(i.hasNext());
142        assertFalse(c == i.next());
143    
144        final List<Throwable> list = Throwables.toList(c);
145        assertNotNull(list);
146        assertEquals(3, list.size());
147        assertSame(c, list.get(0));
148      }
149    
150      private static final class ThrowableCollection extends Throwable implements Iterable<Throwable> {
151    
152        /**
153         * A serial version identifier uniquely identifying the version of
154         * this class.  See the <a
155         * href="http://download.oracle.com/javase/6/docs/api/java/io/Serializable.html">documentation
156         * for the {@code Serializable} class</a> for details.
157         */
158        private static final long serialVersionUID = 1L;
159    
160        private final List<Throwable> list;
161    
162        private ThrowableCollection() {
163          super();
164          this.list = new ArrayList<Throwable>();
165          this.list.add(new Exception("first affiliate"));
166          this.list.add(new Exception("second affiliate"));
167        }
168    
169        @Override
170        public Iterator<Throwable> iterator() {
171          return this.list.iterator();
172        }
173    
174      }
175
