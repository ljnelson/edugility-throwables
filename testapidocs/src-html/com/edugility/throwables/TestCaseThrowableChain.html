028    package com.edugility.throwables;
029    
030    import java.util.Iterator;
031    
032    import org.junit.After;
033    import org.junit.Before;
034    import org.junit.Test;
035    
036    import static org.junit.Assert.assertEquals;
037    import static org.junit.Assert.assertNotNull;
038    import static org.junit.Assert.assertSame;
039    import static org.junit.Assert.assertTrue;
040    import static org.junit.Assert.assertFalse;
041    
042    public class TestCaseThrowableChain {
043    
044      private ThrowableChain chain;
045    
046      private Exception expectedCause;
047      
048      @Before
049      public void setUp() throws Exception {
050        this.chain = new ThrowableChain();
051        this.expectedCause = new Exception("1");
052        this.chain.add(this.expectedCause);
053        assertSame(this.expectedCause, this.chain.getCause());
054        this.chain.add(new Exception("2"));
055        this.chain.add(new Exception("3"));
056      }
057    
058      @Test
059      public void testCause() {
060        assertNotNull(this.chain.getCause());
061        assertEquals("1", this.chain.getCause().getMessage());
062        assertEquals(3, this.chain.size());
063      }
064    
065      @Test
066      public void testIteration() {
067        final Iterator<Throwable> i = this.chain.iterator();
068        assertNotNull(i);
069        assertTrue(i.hasNext());
070        Throwable t = i.next();
071    
072        // The first item in the iteration is always the ThrowableChain
073        // itself.
074        assertSame(this.chain, t);
075    
076        assertTrue(i.hasNext());
077        t = i.next();
078        assertNotNull(t);
079    
080        // The next item is NOT the first exception added (that becomes
081        // the ThrowableChain's cause).  It is instead the SECOND
082        // exception added.
083        assertEquals("2", t.getMessage());
084        assertEquals("1", this.chain.getCause().getMessage());
085    
086        t = i.next();
087        assertNotNull(t);
088    
089        // From that point forward the iteration proceeds normally.
090        assertEquals("3", t.getMessage());
091      }
092    
093      @Test
094      public void testOddCauseAndAddSituations() throws Exception {
095        ThrowableChain chain = new ThrowableChain();
096        assertEquals(1, chain.size());
097        assertTrue(chain.getCause() == null);
098    
099        // Initializing a chain's cause must not actually add that cause
100        // to the list or alter the chain's size.
101        final Exception cause = new Exception("cause");
102        chain.initCause(cause);
103        assertSame(cause, chain.getCause());
104        assertEquals(1, chain.size());
105        assertFalse(chain.asList().contains(cause));
106        
107        // Removal has no effect on the cause.  Once a cause is installed,
108        // it is there forever.
109        assertFalse(chain.remove(cause));
110        assertEquals(1, chain.size());
111        assertSame(cause, chain.getCause());
112    
113        chain = new ThrowableChain();
114        
115        // Adding the first item initializes the cause but does not affect
116        // the size/contents of the chain's affiliates.
117        assertFalse(chain.add(cause));
118        assertEquals(1, chain.size());
119        assertFalse(chain.asList().contains(cause));
120        assertSame(cause, chain.getCause());
121    
122        // Adding subsequent items affects the list but does not affect
123        // the cause.
124        final Exception affiliate = new Exception("affiliate");
125        chain.add(affiliate);
126        assertSame(cause, chain.getCause());
127        assertEquals(2, chain.size());
128        assertFalse(chain.asList().contains(cause));
129    
130        // Adding an exception that is already present has no effect.
131        assertFalse(chain.add(affiliate));
132    
133      }
